1. Условный оператор if и функции

Создайте функцию, которая принимает число и проверяет, является ли оно положительным, отрицательным или нулем.

function checkNumber(number) {
    if (number > 0) {
        return "Число положительное";
    } else if (number < 0) {
        return "Число отрицательное";
    } else {
        return "Число равно нулю";
    }
}


console.log(checkNumber(10));
console.log(checkNumber(-5));
console.log(checkNumber(0));

3. Классы и объекты

Создайте класс Person с свойствами name и age и методом для вывода информации.

class Person {
    constructor(name, age) {
        this.name = name;
        this.age = age;
    }
    
    getInfo() {
        return `Имя: ${this.name}, Возраст: ${this.age}`;
    }
}


const person1 = new Person("Иван", 25);
const person2 = new Person("Мария", 30);

console.log(person1.getInfo());
console.log(person2.getInfo());
